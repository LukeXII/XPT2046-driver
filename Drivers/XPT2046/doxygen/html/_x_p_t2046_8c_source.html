16void XPT2046_setControllerConfig(adcRes_t res, adcMode_t mode, powerMode_t pd, uint16_t rxplate)
17{
18 tsHandler.res = res;
19 tsHandler.mode = mode;
20 tsHandler.powerMode = pd;
21 tsHandler.RxPlate = rxplate;
22
23 CTRL_BYTE_LOW = (res << 3) | (mode << 2) | pd;

28uint16_t XPT2046_getCoord(requestedData_t coord)
29{
30 uint8_t receivedData[3];
31 unsigned char controlByte;
32 uint16_t coordData;
33
34 receivedData[2] = '\0';
35
36 if(!isControllerConfigured) // Si el controlador no está configurado, aborta la operación
37 return -1;
38
39 XPT2046_chipEnable();
40
41 // Arma el byte de control segun el dato a consultar
42 switch(coord)
43 {
44 case COORD_X:
45 controlByte = (CTRL_BYTE_HIGH_X << 4) | CTRL_BYTE_LOW;
46 break;
47
48 case COORD_Y:
49 controlByte = (CTRL_BYTE_HIGH_Y << 4) | CTRL_BYTE_LOW;
50 break;
51
52 case COORD_Z1:
53 controlByte = (CTRL_BYTE_HIGH_Z1 << 4) | CTRL_BYTE_LOW;
54 break;
55
56 case COORD_Z2:
57 controlByte = (CTRL_BYTE_HIGH_Z2 << 4) | CTRL_BYTE_LOW;
58 break;
59
60 case TEMP0:
61 controlByte = CTRL_BYTE_TEMP0;
62 break;
63
64 case TEMP1:
65 controlByte = CTRL_BYTE_TEMP1;
66 break;
67 }
68
69 // Envia el byte de control y luego recibe los dos bytes de la respuesta
70 XPT2046_startConversion(controlByte);
71 XPT2046_getConversionData(receivedData, 16);
72 XPT2046_chipDisable();
73
74 // Pone los bits de la respuesta en el orden correcto
75 coordData = XPT2046_convertReceivedData(receivedData[0], receivedData[1]);
76
77 return coordData;
78}

80uint16_t XPT2046_getPressure(void)
81{
82 uint16_t coordX, coordZ1, coordZ2;
83
84 if(!isControllerConfigured)
85 return -1;
86
87 // Obtiene las coordenadas X, Z1 y Z2 del controlador para poder calcular la presión
88 coordX = XPT2046_getCoord(COORD_X);
89 coordZ1 = XPT2046_getCoord(COORD_Z1);
90 coordZ2 = XPT2046_getCoord(COORD_Z2);
91
92 return tsHandler.RxPlate*coordX*(coordZ2/coordZ1 - 1)/MAX_12BIT_RES;
93}

95float XPT2046_getTemp(void)
96{
97 uint16_t temp0, temp1;
98
99 if(!isControllerConfigured)
100 return -1;
101
102 // Obtiene los valores de TEMP0 y TEMP1 del controlador
103 temp0 = XPT2046_getCoord(TEMP0);
104 temp1 = XPT2046_getCoord(TEMP1);
105
106 return 2.573*((float)temp1 - (float)temp0)*2500/4095;
107}
